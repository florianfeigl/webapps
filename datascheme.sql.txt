CREATE TABLE horses (
  id SERIAL PRIMARY KEY,
  name VARCHAR(255),
  sex CHAR(1),
  birth DATE,
  color VARCHAR(50),
  region VARCHAR(255),
  pedigree_id INTEGER,
  breeding_id INTEGER,
  FOREIGN KEY (pedigree_id) REFERENCES pedigree(id),
  FOREIGN KEY (breeding_id) REFERENCES breeding(id)
);

CREATE TABLE pedigree (
  id SERIAL PRIMARY KEY,
  name_id INTEGER,
  sire_id INTEGER,
  dam_id INTEGER,
  damsire_id INTEGER,
  FOREIGN KEY (name_id) REFERENCES horses(id),
  FOREIGN KEY (sire_id) REFERENCES horses(id),
  FOREIGN KEY (dam_id) REFERENCES horses(id),
  FOREIGN KEY (damsire_id) REFERENCES horses(id)
);

CREATE TABLE breeding (
  id SERIAL PRIMARY KEY,
  year DATE,
  dam_id INTEGER,
  sire_id INTEGER,
  attribute VARCHAR(50),
  offspring_id INTEGER,
  FOREIGN KEY (dam_id) REFERENCES horses(id),
  FOREIGN KEY (sire_id) REFERENCES horses(id),
  FOREIGN KEY (offspring_id) REFERENCES horses(id)
);
